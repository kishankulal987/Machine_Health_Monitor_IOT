// ─── Helpers ───
const now = () => new Date().toLocaleTimeString('en-IN', { hour12:false });
const pct = (v, max) => Math.min(100, Math.max(0, (Math.abs(v)/max)*100)).toFixed(1)+'%';

let logCount = 0;
const logList = document.getElementById('log-list');
const eventLogEntries = []; // store all log entries for report

function addLog(msg, type='ok') {
  if (logCount===0) logList.innerHTML='';
  const timestamp = now();
  // Store for report
  eventLogEntries.unshift({ time: timestamp, msg, type });
  if (eventLogEntries.length > 200) eventLogEntries.pop();

  const li = document.createElement('li');
  li.className='log-item';
  li.innerHTML=`<span class="log-time">${timestamp}</span><span class="log-msg ${type}">${msg}</span>`;
  logList.prepend(li);
  logCount++;
  while (logList.children.length > 30) logList.removeChild(logList.lastChild);
}

function updateCard(cardId, valId, tagId, barId, value, threshold, decimals) {
  const card=document.getElementById(cardId);
  const valEl=document.getElementById(valId);
  const tagEl=document.getElementById(tagId);
  const barEl=document.getElementById(barId);
  valEl.textContent = parseFloat(value).toFixed(decimals);
  barEl.style.width = pct(value, threshold*1.5);
  card.className='card'; valEl.className='card-value';
  tagEl.className='alert-tag'; barEl.style.background='var(--cyan)';
  if (Math.abs(value)>=threshold) {
    card.classList.add('alert-critical'); valEl.classList.add('crit');
    tagEl.classList.add('crit'); tagEl.textContent='⚠ THRESHOLD EXCEEDED';
    barEl.style.background='var(--red)';
  } else if (Math.abs(value)>=threshold*0.7) {
    card.classList.add('alert-warning'); valEl.classList.add('warn');
    tagEl.classList.add('warn'); tagEl.textContent='△ APPROACHING LIMIT';
    barEl.style.background='var(--yellow)';
  } else {
    tagEl.textContent='✓ NOMINAL';
  }
}

function updateAxisBar(barId, valId, val, range, color) {
  const p = Math.min(100, Math.max(0, 50+(val/range)*50));
  document.getElementById(barId).style.width = p+'%';
  document.getElementById(barId).style.background = color||'var(--cyan)';
  document.getElementById(valId).textContent = val.toFixed(2);
}

// ─── CRITICAL OVERLAY LOGIC ───
let overlayShowing = false;

function showCriticalOverlay(data) {
  if (overlayShowing) return;
  overlayShowing = true;
  const faults = [];
  if (data.temperature > CRITICAL_THRESH.temp)    faults.push(`TEMP  ${data.temperature.toFixed(1)}°C  [LIMIT: ${CRITICAL_THRESH.temp}°C]`);
  if (data.current     > CRITICAL_THRESH.current) faults.push(`CURRENT  ${data.current.toFixed(2)}A  [LIMIT: ${CRITICAL_THRESH.current}A]`);
  if (data.vibration   > CRITICAL_THRESH.vib)     faults.push(`VIBRATION  ${data.vibration.toFixed(3)}g  [LIMIT: ${CRITICAL_THRESH.vib}g]`);
  if (faults.length === 0) faults.push('THRESHOLD BREACHED — SEE SENSOR DATA');
  document.getElementById('crit-reason').innerHTML =
    '<strong style="color:rgba(255,51,85,0.8);letter-spacing:2px;">FAULT DETECTED:</strong><br><br>' +
    faults.join('<br>');
  document.getElementById('btn-restart').disabled = false;
  document.getElementById('restart-confirm').textContent = '';
  document.getElementById('critical-overlay').classList.add('show');
  addLog('CRITICAL: Machine auto-shutdown triggered.', 'crit');
}

function hideCriticalOverlay() {
  overlayShowing = false;
  document.getElementById('critical-overlay').classList.remove('show');
}

// ─── RESTART COMMAND → Firebase ───
function sendRestartCommand() {
  const btn = document.getElementById('btn-restart');
  const confirmEl = document.getElementById('restart-confirm');
  btn.disabled = true;
  confirmEl.textContent = 'SENDING RESTART COMMAND...';
  ctrlRef.update({ command: 'restart', machine_on: false })
    .then(() => {
      confirmEl.textContent = '✓ COMMAND SENT — WAITING FOR ESP32...';
      addLog('Restart command sent from webpage.', 'warn');
      const listener = ctrlRef.child('machine_on').on('value', (snap) => {
        if (snap.val() === true) {
          ctrlRef.child('machine_on').off('value', listener);
          confirmEl.textContent = '✓ MACHINE RESTARTED SUCCESSFULLY';
          addLog('Machine restarted — back online.', 'ok');
          setTimeout(() => hideCriticalOverlay(), 1500);
        }
      });
      setTimeout(() => {
        ctrlRef.child('machine_on').off('value', listener);
        if (overlayShowing) {
          btn.disabled = false;
          confirmEl.textContent = '⚠ NO RESPONSE — CHECK ESP32 CONNECTION';
        }
      }, 30000);
    })
    .catch(err => {
      confirmEl.textContent = '❌ ERROR: ' + err.message;
      btn.disabled = false;
    });
}

function downloadLogReport() {
  const reportDate = new Date().toLocaleString('en-IN', { hour12: false });
  const d = latestData;

  // ── Helper: status badge text
  function getStatusText(val, thresh, critThresh) {
    if (val >= critThresh) return 'CRITICAL';
    if (val >= thresh)     return 'WARNING';
    if (val >= thresh*0.7) return 'APPROACHING LIMIT';
    return 'NOMINAL';
  }

  // ── Uptime string
  let uptimeStr = '--';
  if (d && d.uptime_sec) {
    const h = Math.floor(d.uptime_sec/3600);
    const m = Math.floor((d.uptime_sec%3600)/60);
    const s = d.uptime_sec%60;
    uptimeStr = `${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  }

  // ── Sensor snapshot
  const temp    = d ? (d.temperature||0) : null;
  const humid   = d ? (d.humidity||0)    : null;
  const current = d ? (d.current||0)     : null;
  const vib     = d ? (d.vibration||0)   : null;
  const aX = d?.accel?.x||0, aY = d?.accel?.y||0, aZ = d?.accel?.z||0;
  const gX = d?.gyro?.x||0,  gY = d?.gyro?.y||0,  gZ = d?.gyro?.z||0;
  const systemStatus = d ? (d.status||'UNKNOWN').toUpperCase() : 'NO DATA';

  // ── Build CSV-style log entries
  const logCSV = ['TIMESTAMP,LEVEL,MESSAGE'];
  for (const entry of eventLogEntries) {
    const level = entry.type === 'crit' ? 'CRITICAL' : entry.type === 'warn' ? 'WARNING' : 'INFO';
    logCSV.push(`${entry.time},${level},"${entry.msg.replace(/"/g,'""')}"`);
  }

  // ── Trend data
  const trendCSV = ['TIMESTAMP,TEMPERATURE_C,CURRENT_A,VIBRATION_G'];
  for (let i = 0; i < chartLabels.length; i++) {
    trendCSV.push(`${chartLabels[i]},${tempData[i]||''},${currentData[i]||''},${vibData[i] !== undefined ? (vibData[i]/10).toFixed(4) : ''}`);
  }

  // ── Fault analysis
  const faults = [];
  if (temp !== null && temp > CRITICAL_THRESH.temp)       faults.push(`Temperature ${temp.toFixed(1)}°C exceeds critical limit of ${CRITICAL_THRESH.temp}°C`);
  if (current !== null && current > CRITICAL_THRESH.current) faults.push(`Current ${current.toFixed(2)}A exceeds critical limit of ${CRITICAL_THRESH.current}A`);
  if (vib !== null && vib > CRITICAL_THRESH.vib)           faults.push(`Vibration ${vib.toFixed(3)}g exceeds critical limit of ${CRITICAL_THRESH.vib}g`);

  // ════ BUILD REPORT TEXT ════
  const divider  = '='.repeat(72);
  const sdivider = '-'.repeat(72);

  const lines = [
    divider,
    '  ESP32 INDUSTRIAL MACHINE HEALTH MONITOR — SYSTEM REPORT',
    divider,
    `  Generated   : ${reportDate}`,
    `  System      : ESP32 + DHT11 + MPU9255 + ACS712 (30A)`,
    `  Firebase DB : esp32-iot-15cfb-default-rtdb.firebaseio.com`,
    `  Uptime      : ${uptimeStr}`,
    `  Status      : ${systemStatus}`,
    divider,
    '',
    '[ 1. SENSOR SNAPSHOT — LATEST READING ]',
    sdivider,
    '',
    `  Temperature      : ${temp !== null ? temp.toFixed(1)+'°C' : 'N/A'}` +
      `  ${temp !== null ? '[ '+getStatusText(temp, THRESH.temp, CRITICAL_THRESH.temp)+' ]' : ''}`,
    `    Warning Threshold  : ${THRESH.temp}°C`,
    `    Critical Threshold : ${CRITICAL_THRESH.temp}°C`,
    '',
    `  Humidity         : ${humid !== null ? humid.toFixed(1)+'%' : 'N/A'}` +
      `  ${humid !== null ? '[ '+getStatusText(humid, THRESH.humid, 999)+' ]' : ''}`,
    `    Warning Threshold  : ${THRESH.humid}%`,
    '',
    `  Current Draw     : ${current !== null ? current.toFixed(2)+'A' : 'N/A'}` +
      `  ${current !== null ? '[ '+getStatusText(current, THRESH.current, CRITICAL_THRESH.current)+' ]' : ''}`,
    `    Warning Threshold  : ${THRESH.current}A`,
    `    Critical Threshold : ${CRITICAL_THRESH.current}A`,
    '',
    `  Vibration Mag.   : ${vib !== null ? vib.toFixed(3)+'g' : 'N/A'}` +
      `  ${vib !== null ? '[ '+getStatusText(vib, THRESH.vib, CRITICAL_THRESH.vib)+' ]' : ''}`,
    `    Warning Threshold  : ${THRESH.vib}g`,
    `    Critical Threshold : ${CRITICAL_THRESH.vib}g`,
    '',
    sdivider,
    '',
    '[ 2. IMU DETAILS — MPU9255 ]',
    sdivider,
    '',
    '  Accelerometer (g):',
    `    X = ${aX.toFixed(4)}    Y = ${aY.toFixed(4)}    Z = ${aZ.toFixed(4)}`,
    '',
    '  Gyroscope (°/s):',
    `    X = ${gX.toFixed(4)}    Y = ${gY.toFixed(4)}    Z = ${gZ.toFixed(4)}`,
    '',
    sdivider,
    '',
    '[ 3. FAULT ANALYSIS ]',
    sdivider,
    '',
    faults.length > 0
      ? faults.map(f => '  !! FAULT : ' + f).join('\n')
      : '  No active faults detected at time of report.',
    '',
    sdivider,
    '',
    '[ 4. EVENT LOG ]',
    sdivider,
    '',
    ...(eventLogEntries.length > 0
      ? eventLogEntries.map(e => {
          const level = e.type === 'crit' ? 'CRITICAL' : e.type === 'warn' ? 'WARNING ' : 'INFO    ';
          return `  [${e.time}]  ${level}  ${e.msg}`;
        })
      : ['  No events recorded in this session.']),
    '',
    sdivider,
    '',
    '[ 5. TREND DATA — LAST 20 READINGS ]',
    sdivider,
    '',
    '  (Full CSV appended at end of file)',
    `  Points captured  : ${chartLabels.length}`,
    temp !== null
      ? `  Temp range       : ${Math.min(...tempData).toFixed(1)}°C — ${Math.max(...tempData).toFixed(1)}°C`
      : '',
    current !== null
      ? `  Current range    : ${Math.min(...currentData).toFixed(2)}A — ${Math.max(...currentData).toFixed(2)}A`
      : '',
    '',
    divider,
    '  END OF REPORT',
    divider,
    '',
    '',
    '[ APPENDIX A — EVENT LOG CSV ]',
    sdivider,
    logCSV.join('\n'),
    '',
    '',
    '[ APPENDIX B — TREND DATA CSV ]',
    sdivider,
    trendCSV.join('\n'),
    '',
  ];

  const reportText = lines.join('\n');

  // ── Trigger download
  const blob = new Blob([reportText], { type: 'text/plain;charset=utf-8' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  const ts   = new Date().toISOString().replace(/[:.]/g,'-').slice(0,19);
  a.href     = url;
  a.download = `machine_health_report_${ts}.txt`;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  addLog('System report downloaded.', 'ok');
}

// ─── Main Firebase Listener ───
let prevStatus = null;
let firstLoad  = true;

dataRef.on('value', (snapshot) => {
  const d = snapshot.val();
  if (!d) return;

  latestData = d; // store for report

  if (firstLoad) {
    document.getElementById('loading').classList.add('hide');
    firstLoad = false;
    addLog('Firebase connected. Receiving data.', 'ok');
  }

  document.getElementById('conn-dot').className  = 'dot online';
  document.getElementById('conn-status').textContent = 'LIVE';

  const temp    = d.temperature || 0;
  const humid   = d.humidity    || 0;
  const current = d.current     || 0;
  const vib     = d.vibration   || 0;
  const aX = (d.accel?.x) || 0;
  const aY = (d.accel?.y) || 0;
  const aZ = (d.accel?.z) || 0;
  const gX = (d.gyro?.x)  || 0;
  const gY = (d.gyro?.y)  || 0;
  const gZ = (d.gyro?.z)  || 0;
  const status  = (d.status || 'NORMAL').toUpperCase();
  const uptime  = d.uptime_sec || 0;

  if (status === 'CRITICAL' && !overlayShowing) {
    showCriticalOverlay({ temperature:temp, current, vibration:vib });
  } else if (status !== 'CRITICAL' && overlayShowing) {
    hideCriticalOverlay();
  }

  updateCard('card-temp',    'val-temp',    'tag-temp',    'bar-temp',    temp,    THRESH.temp,    1);
  updateCard('card-humid',   'val-humid',   'tag-humid',   'bar-humid',   humid,   THRESH.humid,   1);
  updateCard('card-current', 'val-current', 'tag-current', 'bar-current', current, THRESH.current, 2);
  updateCard('card-vib',     'val-vib',     'tag-vib',     'bar-vib',     vib,     THRESH.vib,     3);
  document.getElementById('sub-humidity').textContent = `HUM: ${humid.toFixed(1)} %`;

  updateAxisBar('ab-ax','av-ax', aX, 2, 'var(--cyan)');
  updateAxisBar('ab-ay','av-ay', aY, 2, 'var(--cyan)');
  updateAxisBar('ab-az','av-az', aZ, 2, 'var(--cyan)');
  updateAxisBar('ab-gx','av-gx', gX, 250, 'var(--green)');
  updateAxisBar('ab-gy','av-gy', gY, 250, 'var(--green)');
  updateAxisBar('ab-gz','av-gz', gZ, 250, 'var(--green)');

  const banner=document.getElementById('status-banner');
  const sText =document.getElementById('status-text');
  const sIcon =document.getElementById('status-icon');
  const sMet  =document.getElementById('status-metrics');
  banner.className='status-banner'; sText.className='status-indicator';

  if (status==='CRITICAL') {
    banner.classList.add('critical'); sText.classList.add('critical');
    sText.textContent='⛔ CRITICAL — MACHINE SHUT DOWN'; sIcon.textContent='🚨';
  } else if (status==='WARNING') {
    banner.classList.add('warning'); sText.classList.add('warning');
    sText.textContent='⚠ WARNING — THRESHOLD BREACHED'; sIcon.textContent='⚠️';
  } else {
    banner.classList.add('normal'); sText.classList.add('normal');
    sText.textContent='✅ NORMAL OPERATION'; sIcon.textContent='✅';
  }

  sMet.textContent=`TEMP ${temp.toFixed(1)}°C · HUMID ${humid.toFixed(0)}% · ${current.toFixed(2)}A · VIB ${vib.toFixed(3)}g`;

  const h=Math.floor(uptime/3600), m=Math.floor((uptime%3600)/60), s=uptime%60;
  document.getElementById('uptime-display').textContent=
    `UPTIME: ${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;

  if (status !== prevStatus) {
    if (status==='CRITICAL')     addLog('CRITICAL: Auto-shutdown — relay cut.', 'crit');
    else if (status==='WARNING') addLog('WARNING: Threshold breached.', 'warn');
    else if (prevStatus!==null)  addLog('System back to normal operation.', 'ok');
    prevStatus = status;
  }

  chartLabels.push(now());
  tempData.push(+temp.toFixed(1));
  currentData.push(+current.toFixed(2));
  vibData.push(+(vib*10).toFixed(3));
  if (chartLabels.length>MAX_PTS) { chartLabels.shift(); tempData.shift(); currentData.shift(); vibData.shift(); }
  chart.update();

  document.getElementById('last-update').textContent=`LAST UPDATE: ${now()}`;
});
